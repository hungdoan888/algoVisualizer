Time: O(n2) | Space: O(1)

def selectionSort(array):

for i in range(len(array)):

smallestIdx = i

for j in range(i + 1, len(array)):

if array[j] < array[smallestIdx]:

smallestIdx = j

array[i], array[smallestIdx] = array[smallestIdx], array[i]

return array
